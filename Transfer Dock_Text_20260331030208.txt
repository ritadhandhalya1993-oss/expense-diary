<script>
let data = JSON.parse(localStorage.getItem("financeData")) || [];

function save() {
  localStorage.setItem("financeData", JSON.stringify(data));
}

function checkOther() {
  customType.style.display = (expenseType.value === "Other") ? "block" : "none";
}

function addData() {
  let type = expenseType.value;
  if(type === "Other") type = customType.value || "Other";

  data.push({
    date: date.value,
    source: incomeSource.value,
    amount: Number(amount.value),
    expense: Number(expense.value),
    type: type
  });

  save();
  render();
}

function render() {
  let totalIncome = 0, totalExp = 0;
  let category = {};

  let month = monthFilter.value;
  let searchVal = search.value.toLowerCase();

  tableData.innerHTML = "";

  data.forEach(d => {

    if(month && !d.date.startsWith(month)) return;

    if(searchVal &&
      !d.source.toLowerCase().includes(searchVal) &&
      !d.type.toLowerCase().includes(searchVal)) return;

    totalIncome += d.amount;
    totalExp += d.expense;

    if(!category[d.type]) category[d.type]=0;
    category[d.type]+=d.expense;

    tableData.innerHTML += `
    <tr>
      <td>${d.date}</td>
      <td>${d.source}</td>
      <td>${d.amount}</td>
      <td>${d.expense}</td>
      <td>${d.type}</td>
    </tr>`;
  });

  totalExpense.innerText = totalExp;
  balance.innerText = totalIncome - totalExp;

  updateCharts(totalIncome, totalExp, category);
}

let incomeChart = new Chart(document.getElementById("incomeExpenseChart"), {
  type: 'bar',
  data: { labels: ['Income','Expense'], datasets:[{data:[0,0]}] }
});

let categoryChart = new Chart(document.getElementById("categoryChart"), {
  type: 'bar',
  data: { labels:[], datasets:[{data:[]}] }
});

function updateCharts(income, expense, category) {
  incomeChart.data.datasets[0].data = [income, expense];
  incomeChart.update();

  categoryChart.data.labels = Object.keys(category);
  categoryChart.data.datasets[0].data = Object.values(category);
  categoryChart.update();
}

function exportCSV() {
  let csv = "Date,Source,Income,Expense,Type\n";
  data.forEach(d => {
    csv += `${d.date},${d.source},${d.amount},${d.expense},${d.type}\n`;
  });

  let blob = new Blob([csv]);
  let a = document.createElement("a");
  a.href = URL.createObjectURL(blob);
  a.download = "finance.csv";
  a.click();
}

function backupData() {
  let blob = new Blob([JSON.stringify(data)]);
  let a = document.createElement("a");
  a.href = URL.createObjectURL(blob);
  a.download = "backup.json";
  a.click();
}

function restoreData(e) {
  let file = e.target.files[0];
  let reader = new FileReader();
  reader.onload = function() {
    data = JSON.parse(reader.result);
    save();
    render();
  };
  reader.readAsText(file);
}

render();
</script>
